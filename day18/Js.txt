 저장방식
1.변수: 변할 수 있는 수
    let 변수명= 값; 
    let variable = 0;

 2.상수: 변하지 않는 수
    const 상수명 = 값; 
    const constant = 0;

자료형 종류
 1. 숫자: 숫자들로만 이루어진 자료형
    let 변수명 = 10; 
    let number = 10;

 2. 문자열: 문자들의 집합(숫자도 문자에 포함됨)
    const 상수명 = '문자열' 
    const string = 'number'

 3. 불: 맞는지 아닌지를 (true or flase)로 표현하는 자료형
    true or false;
    let boolean =  3>2;

 4. 배열: 여러 데이터들을 묶어서 순서대로 번호를 부여하여 활용하는 자료형
    let 변수명 = []; 
    let array = [1,2,3];

 5. 함수: 기능을 구현시킨 코드들의 집합
    function 함수명(){} 
    function fuc( ){console.log('함수')}; fuc();

 6. 객체: 자료들을 묶어놓은 것
    let 객체명 = {key: value};
    let object = {객체: object};
